# DML
## 데이터 조작어
### 종류 : SELECT, INSERT, UPDATE, DELETE

## SELECT
: 데이터 조회

# 기본 문법
SELECT 컬럼명1, 컬럼명2, ... ( 전체 * )
FROM 테이블명
WHERE 조건
GROUP BY 그룹기준 컬럼... HAVING 그룹 조건
ORDER BY 정렬기준 컬럼... [ASC / DESC];

* 필수 키워드 : SELECT, FROM
* 선택 키워드 : WHERE, GROUP BY, ORDER BY
* 유의사항
    - 위에서 작성한 순서대로 키워드를 작성하지 않으면 오류 발생!

* 실행 순서
    - FROM -> WHERE -> GROUP BY -> HAVING -> SELECT -> ORDER BY
    1. 테이블 선택
    2. 조건에 맞는 데이터 선택
    3. 그룹 기준 지정
    4. 그룹별로 그룹조건에 맞는 데이터를 선택
    EX) (사원 수가 5명 이상인 그룹)만 부서코드를 기준으로 급여 합계 구하기
    5. 조회할 컬럼 선택
    6. 조회된 결과를 정렬기준에 따라 정렬

# 연산자
- 산술 연산자 : 더하기, 빼기, 곱하기, 나누기
- 비교 연산자 : A > B | A < B | A >= B | A <= B | A = B | A != B | A <> B (다름) | A ^= B (다름)
- BETWEEN A AND B : X >= A AND X <= B
- 논리 연산자 : AND, OR, NOT
- IN         : 특정 열에 포함되는 여러 값을 조회할 때
                * A IN (값1, 값2, 값3, ... ) : A 속성의 값을 1, 2, 3 을 포함하여 조회
                * A NOT IN (값1, 값2, 값3, ... ) : A 속성의 값을 1, 2, 3 을 제외하여 조회
- LIKE      
            * A LIKE '_' -> 한 문자를 대체하는 와일드 카드
            * A LIKE '%' -> 공백포함 여러 글자를 대체하는 와일드 카드
            EX) 이름 LIKE '김%' : 성이 김씨인 사람 조회
- IS NULL    : 열의 값 존재 여부를 확인하는 연산
                * A IS NULL : A 가 NULL 일 때
                * A IS NOT NULL : A 가 NULL 아닐 때
    * NULL 유의사항
    - NULL 과의 연산 : NULL 과 연산한 데이터는 NULL 된다
        NULL + 10 = NULL
        NULL > 10 = NULL
    - NULL 과 연산되는 경우 방지하기 위한 방법
        NVL(), NVL2() 함수 이용
- 집합 연산자
    * UNION     : 합집합, A UNION B, A 와 B 의 결과를 합집합으로 묶는다. 중복 데이터 제거
    * UNION ALL : 합집합, A UNION B, A 와 B 의 결과를 합집합으로 묶는다. 중복 데이터 포함
    * MINUS     : 차집합, A MINUS B, A 와 B 의 결과를 차집합으로 출력
    * INTERSECT : 교집한, A INTERSECT B, A 와 B 의 결과를 교집합으로 출력

* 연산자 우선순위
    1. () -> 2.*,/ -> 3.+,- -> 4.비교 -> 5.IS NULL, LIKE, IN -> 6.BETWEEN A AND B -> 7.NOT -> 8.AND -> 9.OR

## INSERT
: 데이터 추가

# 기본문법
INSERT INTO 테이블명 ( 컬럼1, 컬럼2, 컬럼3, ...)
VALUES (값1, 값2, 값3, ...);
* 컬럼 작성 순서와, 값 작성 순서를 짝을 맞춰서 작성해야한다

# 컬럼명 생략
INSERT INTO 테이블명 
VALUES (값1, 값2, 값3, ...);
* 컬럼명 생략하고 작성하면, 컬럼 정의 순서대로 모든 값을 지정해주어야한다.

# 데이터 조회하여 추가하기
INSERT INTO 테이블명 (컬럼1, 컬럼2, 컬럼3, ...)
SELECT 컬럼1, 컬럼2, 컬럼3, ...
FROM 테이블
[WHERE 조건];

## UPDATE
: 데이터 수정

# 기본문법
UPDATE 테이블명
    SET 컬럼1 = 변경할 값,
        컬럼2 = 변경할 값,
        ...
[WHERE 조건];

# 데이터 조회하여 수정하기
UPDATE 테이블명
    SET (컬럼1, 컬럼2, ...) = (SELECT 컬럼1, 컬럼2, ...
                                FROM 테이블명2
                                WHERE 조건
                            )
[WHERE 조건];

## DELETE
: 데이터 삭제
DELETE FROM 테이블
[WHERE 조건];
